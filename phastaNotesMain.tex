\section{Shorthand}
    \begin{equation}
        \phi_{,t} = \frac{\partial \phi}{\partial t}
    \end{equation}

    \begin{equation}
        \phi_{,i} = \frac{\partial \phi}{\partial x_i}
    \end{equation}

    \begin{equation}
        u_{i,t} = \frac{\partial u_i}{\partial t}
    \end{equation}

    \begin{equation}
        u_{i,j} = \frac{\partial u_i}{\partial x_j}
    \end{equation}

    \begin{equation}
        \left[\phi u_i \right]_{,j} = \frac{\partial \phi u_i}{\partial x_j}
    \end{equation}
